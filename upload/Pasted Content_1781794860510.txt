### Auth Store (Zustand + persist)

`src/lib/auth-store.ts`:

```typescript
'use client';

import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { demoUsers, type DemoUser } from '@/data/demo-users';

interface AuthState {
  user: DemoUser | null;
  isAuthenticated: boolean;
  login: (email: string, password: string) => { success: boolean; error?: string };
  loginAs: (userId: string) => void; // quick demo switch
  register: (data: { name: string; email: string; password: string; interests: string[] }) => { success: boolean; error?: string };
  logout: () => void;
  updateProfile: (updates: Partial<DemoUser>) => void;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set) => ({
      user: null,
      isAuthenticated: false,
      
      login: (email, password) => {
        const user = demoUsers.find(u => u.email === email);
        if (!user) return { success: false, error: 'user-not-found' };
        if (password !== user.password) return { success: false, error: 'wrong-password' };
        set({ user, isAuthenticated: true });
        return { success: true };
      },
      
      loginAs: (userId) => {
        const user = demoUsers.find(u => u.id === userId);
        if (user) set({ user, isAuthenticated: true });
      },
      
      register: (data) => {
        // mock — just create a new user object (not persisted in demoUsers list)
        const newUser: DemoUser = {
          id: `user-${Date.now()}`,
          email: data.email,
          name: JSON.stringify({ ar: data.name, en: data.name }),
          nameLocalized: { ar: data.name, en: data.name },
          avatarUrl: '/images/avatars/default.jpg', // PROD-ONLY: upload to Cloudinary
          bio: JSON.stringify({ ar: '', en: '' }),
          bioLocalized: { ar: '', en: '' },
          gender: 'male', // default; user can update later
          nationality: '',
          interests: data.interests,
          membershipTier: 'NEWCOMER',
          hostedCount: 0,
          attendedCount: 0,
          password: data.password,
        };
        set({ user: newUser, isAuthenticated: true });
        return { success: true };
      },
      
      logout: () => set({ user: null, isAuthenticated: false }),
      
      updateProfile: (updates) => set(state => ({
        user: state.user ? { ...state.user, ...updates } : null,
      })),
    }),
    {
      name: 'lamma-auth',
      partialize: (state) => ({ user: state.user, isAuthenticated: state.isAuthenticated }),
    }
  )
);
```

### Applications Store (localStorage)

`src/lib/applications-store.ts`:

```typescript
'use client';

import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { z } from 'zod';

export const ApplicationSchema = z.object({
  id: z.string(),
  gatheringSlug: z.string(),
  gatheringTitle: z.string(), // localized JSON for display
  userId: z.string(),
  userEmail: z.string(),
  userName: z.string(), // localized JSON
  motivation: z.string().min(50),
  background: z.string().optional(),
  customAnswers: z.record(z.string(), z.string()).default({}),
  dietaryRestrictions: z.string().optional(),
  accessibilityNeeds: z.string().optional(),
  matchScore: z.number().min(0).max(100).optional(), // mock — random 60-95
  status: z.enum(['PENDING', 'APPROVED', 'REJECTED', 'WAITLISTED', 'ATTENDED', 'NO_SHOW', 'CANCELLED']).default('PENDING'),
  reviewedAt: z.string().optional(),
  reviewedBy: z.string().optional(),
  statusHistory: z.array(z.object({
    status: z.string(),
    timestamp: z.string(),
    note: z.string().optional(),
  })).default([]),
  createdAt: z.string(),
  updatedAt: z.string(),
});

export type Application = z.infer<typeof ApplicationSchema>;

interface ApplicationsState {
  applications: Application[];
  submit: (app: Omit<Application, 'id' | 'createdAt' | 'updatedAt' | 'status' | 'statusHistory' | 'matchScore'>) => Application;
  cancel: (id: string) => void;
  getByUser: (userId: string) => Application[];
  getByGathering: (gatheringSlug: string) => Application[];
  getById: (id: string) => Application | undefined;
  updateStatus: (id: string, status: Application['status'], reviewerId?: string) => void;
}

export const useApplicationsStore = create<ApplicationsState>()(
  persist(
    (set, get) => ({
      applications: [],
      
      submit: (app) => {
        const now = new Date().toISOString();
        const newApp: Application = {
          ...app,
          id: `app-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`,
          status: 'PENDING',
          matchScore: Math.floor(Math.random() * 36) + 60, // 60-95 mock
          statusHistory: [{
            status: 'PENDING',
            timestamp: now,
            note: 'تم استلام الطلب',
          }],
          createdAt: now,
          updatedAt: now,
        };
        set(state => ({ applications: [...state.applications, newApp] }));
        return newApp;
      },
      
      cancel: (id) => {
        set(state => ({
          applications: state.applications.map(app =>
            app.id === id
              ? {
                  ...app,
                  status: 'CANCELLED',
                  updatedAt: new Date().toISOString(),
                  statusHistory: [
                    ...app.statusHistory,
                    { status: 'CANCELLED', timestamp: new Date().toISOString(), note: 'ألغى المستخدم الطلب' },
                  ],
                }
              : app
          ),
        }));
      },
      
      getByUser: (userId) => get().applications.filter(a => a.userId === userId && a.status !== 'CANCELLED'),
      getByGathering: (gatheringSlug) => get().applications.filter(a => a.gatheringSlug === gatheringSlug && a.status !== 'CANCELLED'),
      getById: (id) => get().applications.find(a => a.id === id),
      
      updateStatus: (id, status, reviewerId) => {
        const now = new Date().toISOString();
        set(state => ({
          applications: state.applications.map(app =>
            app.id === id
              ? {
                  ...app,
                  status,
                  reviewedAt: now,
                  reviewedBy: reviewerId,
                  updatedAt: now,
                  statusHistory: [
                    ...app.statusHistory,
                    { status, timestamp: now },
                  ],
                }
              : app
          ),
        }));
      },
    }),
    { name: 'lamma-applications' }
  )
);
```

```tsx
'use client';

import { useState } from 'react';
import { useAuthStore } from '@/lib/auth-store';
import { useTranslations } from 'next-intl';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogDescription, DialogFooter } from '@/components/ui/dialog';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Label } from '@/components/ui/label';
import { Mail, Lock, Users } from 'lucide-react';
import { demoUsers } from '@/data/demo-users';

interface Props {
  open: boolean;
  onOpenChange: (open: boolean) => void;
}

export function LoginDialog({ open, onOpenChange }: Props) {
  const t = useTranslations('auth');
  const { login, loginAs } = useAuthStore();
  const [email, setEmail] = useState('');
  const [password, setPassword] = useState('');
  const [error, setError] = useState<string | null>(null);
  
  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();
    setError(null);
    const result = login(email, password);

      onOpenChange(false);
      setEmail('');
      setPassword('');
    } else {
      setError(t(`errors.${result.error}`));
    }
  };
  
  const handleDemoLogin = (userId: string) => {
    loginAs(userId);
    onOpenChange(false);
  };
  
  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent className="sm:max-w-md">
        <DialogHeader>
          <DialogTitle className="font-display text-2xl">{t('loginTitle')}</DialogTitle>
          <DialogDescription>{t('loginSubtitle')}</DialogDescription>
        </DialogHeader>
        
        <form onSubmit={handleSubmit} className="space-y-4">
          <div className="space-y-2">
```

```tsx
import { useTranslations } from 'next-intl';
import { Clock, CheckCircle2, XCircle, ListPlus, Calendar, UserX, Ban } from 'lucide-react';
import { cn } from '@/lib/utils';

type Status = 'PENDING' | 'APPROVED' | 'REJECTED' | 'WAITLISTED' | 'ATTENDED' | 'NO_SHOW' | 'CANCELLED';

const config: Record<Status, { icon: typeof Clock; className: string }> = {
  PENDING: { icon: Clock, className: 'bg-saffron/15 text-saffron border-saffron/30' },
  APPROVED: { icon: CheckCircle2, className: 'bg-success/15 text-success border-success/30' },
  REJECTED: { icon: XCircle, className: 'bg-error/15 text-error border-error/30' },
  WAITLISTED: { icon: ListPlus, className: 'bg-teal/15 text-teal border-teal/30' },
  ATTENDED: { icon: Calendar, className: 'bg-success/15 text-success border-success/30' },
  NO_SHOW: { icon: UserX, className: 'bg-stone/15 text-stone border-stone/30' },
  CANCELLED: { icon: Ban, className: 'bg-stone/15 text-stone border-stone/30' },
};

export function StatusBadge({ status, size = 'default' }: { status: Status; size?: 'sm' | 'default' }) {
  const t = useTranslations('application_status');
  const { icon: Icon, className } = config[status];
  
  return (
    <span className={cn(
      'inline-flex items-center gap-1.5 px-2.5 py-1 rounded-full border text-xs font-medium',
      className,
      size === 'sm' && 'px-2 py-0.5 text-[10px]'
    )}>
      <Icon className={size === 'sm' ? 'size-3' : 'size-3.5'} />
      {t(status)}
    </span>
  );
}
```

```tsx
import { useAuthStore } from '@/lib/auth-store';
import { useApplicationsStore } from '@/lib/applications-store';
import { useTranslations } from 'next-intl';
import { useRouter } from '@/i18n/routing';
import { Button } from '@/components/ui/button';
import { Card } from '@/components/ui/card';
import { StatusBadge } from '@/components/lamma/applications/StatusBadge';
import { localized } from '@/lib/use-localized';
import { CheckCircle2, LogIn } from 'lucide-react';

interface Props {
  gathering: {
    slug: string;
    title: { ar: string; en: string };
    priceKwd: number;
    isFree: boolean;
    status: string;
    applicationsCloseAt?: Date;
    capacityMax: number;
  };
}

export function ApplicationCTA({ gathering }: Props) {
  const t = useTranslations('gatherings');
  const { user, isAuthenticated } = useAuthStore();
  const applications = useApplicationsStore();
  const router = useRouter();
  
  const existingApp = user
    ? applications.getByUser(user.id).find(a => a.gatheringSlug === gathering.slug)
    : undefined;
  
  const isClosed = gathering.status === 'APPLICATIONS_CLOSED' || 
    (gathering.applicationsCloseAt && new Date(gathering.applicationsCloseAt) < new Date());
  
  const handleApply = () => {
    if (!isAuthenticated) {
      // Show login dialog — use a global event
      window.dispatchEvent(new CustomEvent('lamma:show-login'));
      return;
    }
    router.push(`/gatherings/apply/${gathering.slug}`);
  };
  
  return (
    <Card className="p-6 bg-paper border-clay/20 sticky top-24">
      <div className="flex items-center justify-between mb-4">
        <div>
          <p className="text-sm text-stone">{t('price')}</p>
          <p className="text-3xl font-display tabular">
            {gathering.isFree ? t('free') : `${gathering.priceKwd.toFixed(3)} KWD`}
          </p>
        </div>
        <StatusBadge status={gathering.status as any} />
      </div>
      
      {existingApp ? (
        <div className="space-y-3">
          <div className="bg-success/10 border border-success/20 rounded-lg p-3 flex items-center gap-2">
            <CheckCircle2 className="size-5 text-success flex-shrink-0" />
            <div>
              <p className="text-sm font-medium text-success">{t('alreadyApplied')}</p>
              <p className="text-xs text-stone">{t('applicationId')}: {existingApp.id.slice(-8)}</p>
            </div>
          </div>
          <div className="flex items-center justify-between">
            <span className="text-sm text-stone">{t('currentStatus')}</span>
            <StatusBadge status={existingApp.status} />
          </div>
          <Button
            variant="outline"
            className="w-full"
            onClick={() => router.push(`/dashboard/applications`)}
          >
            {t('viewInDashboard')}
          </Button>
        </div>
      ) : isClosed ? (
        <div className="space-y-3">
          <div className="bg-stone/10 border border-stone/20 rounded-lg p-3">
            <p className="text-sm text-stone">{t('applicationsClosed')}</p>
          </div>
          <Button disabled className="w-full bg-stone">
            {t('closed')}
          </Button>
        </div>
      ) : (
        <>
          <Button
            size="lg"
            className="w-full bg-clay hover:bg-clay/90 text-paper"
            onClick={handleApply}
          >
            {isAuthenticated ? t('applyCta') : (
              <span className="flex items-center gap-2">
                <LogIn className="size-4" />
                {t('loginToApply')}
              </span>
            )}
          </Button>
          
          <p className="text-xs text-stone text-center mt-3">
            {t('applyNote')}
          </p>
          
          {gathering.applicationsCloseAt && (
            <p className="text-xs text-saffron text-center mt-2">
              {t('applicationsCloseIn', { 
                time: formatRelative(gathering.applicationsCloseAt, locale) 
              })}
            </p>
          )}
        </>
      )}
    </Card>
  );
}
```

### Hook for multi-step state

`src/components/lamma/applications/useApplicationForm.ts`:

```typescript
'use client';

import { useState, useEffect } from 'react';
import { useRouter } from '@/i18n/routing';
import { useAuthStore } from '@/lib/auth-store';
import { useApplicationsStore } from '@/lib/applications-store';
import { localized } from '@/lib/use-localized';
import { z } from 'zod';

const motivationSchema = z.string().min(50, 'motivation-too-short');

export function useApplicationForm(gatheringSlug: string) {
  const router = useRouter();
  const { user, isAuthenticated } = useAuthStore();
  const { submit } = useApplicationsStore();
  
  const [step, setStep] = useState(1);
  const [direction, setDirection] = useState<'forward' | 'backward'>('forward');
  const [errors, setErrors] = useState<Record<string, string>>({});
  
  const [data, setData] = useState({
    motivation: '',
    background: '',
    customAnswers: {} as Record<string, string>,
    dietaryRestrictions: '',
    accessibilityNeeds: '',
    emergencyContact: '',
    declaration: false,
  });
  
  // Redirect to login if not authenticated
  useEffect(() => {
    if (!isAuthenticated) {
      window.dispatchEvent(new CustomEvent('lamma:show-login'));
    }
  }, [isAuthenticated]);
  
  const updateField = (field: string, value: any) => {
    setData(prev => ({ ...prev, [field]: value }));
    setErrors(prev => ({ ...prev, [field]: '' }));
  };
  
  const validateStep = (stepNum: number): boolean => {
    const newErrors: Record<string, string> = {};
    
    if (stepNum === 1) {
      const result = motivationSchema.safeParse(data.motivation);
      if (!result.success) {
        newErrors.motivation = 'motivation-too-short';
      }
    }
    
    // Validate custom questions in step 2 (required ones)
    // ... (depends on gathering's applicationQuestions)
    
    if (stepNum === 4 && !data.declaration) {
      newErrors.declaration = 'declaration-required';
    }
    
    setErrors(newErrors);
    return Object.keys(newErrors).length === 0;
  };
  
  const next = () => {
    if (!validateStep(step)) return;
    setDirection('forward');
    setStep(s => Math.min(s + 1, 4));
  };
  
  const back = () => {
    setDirection('backward');
    setStep(s => Math.max(s - 1, 1));
  };
  
  const submitForm = (gathering: any) => {
    if (!validateStep(4) || !user) return;
    
    const newApp = submit({
      gatheringSlug: gathering.slug,
      gatheringTitle: JSON.stringify(gathering.title),
      userId: user.id,
      userEmail: user.email,
      userName: JSON.stringify(user.nameLocalized),
      motivation: data.motivation,
      background: data.background || undefined,
      customAnswers: data.customAnswers,
      dietaryRestrictions: data.dietaryRestrictions || undefined,
      accessibilityNeeds: data.accessibilityNeeds || undefined,
    });
    
    router.push(`/gatherings/apply/${gathering.slug}/success?appId=${newApp.id}`);
  };
  
  return { step, direction, data, errors, updateField, next, back, submitForm, isAuthenticated };
}
```

### Custom questions rendering
كل `gathering.applicationQuestions` array من objects بالشكل:
```typescript
{ id: string, question: { ar, en }, type: 'text' | 'textarea' | 'select', required: boolean, options?: string[] }
```

في Step2CustomQuestions، اعرض كل سؤال ديناميكياً بناءً على `type`:
- `text` → `<Input>`
- `textarea` → `<Textarea>`
- `select` → `<Select>` with options

```tsx
'use client';

import { useEffect } from 'react';
import { useAuthStore } from '@/lib/auth-store';
import { useRouter } from '@/i18n/routing';

export function RequireAuth({ children }: { children: React.ReactNode }) {
  const { isAuthenticated } = useAuthStore();
  const router = useRouter();
  
  useEffect(() => {
    if (!isAuthenticated) {
      window.dispatchEvent(new CustomEvent('lamma:show-login'));
    }
  }, [isAuthenticated]);
  
  if (!isAuthenticated) {
    return (
      <div className="min-h-[60vh] flex items-center justify-center">
        <div className="text-center">
          <p className="text-stone">يجب تسجيل الدخول للوصول إلى هذه الصفحة</p>
        </div>
      </div>
    );
  }
  
  return <>{children}</>;
}
```

### 1. ApplicationCTA (تحقّق + إكمال)

في `src/components/lamma/gatherings/detail/ApplicationCTA.tsx`:

```tsx
'use client';

import { useAuthStore } from '@/lib/auth-store';
import { useApplicationsStore } from '@/lib/applications-store';
import { useTranslations } from 'next-intl';
import { useRouter } from '@/i18n/routing';
import { Button } from '@/components/ui/button';
import { Card } from '@/components/ui/card';
import { StatusBadge } from '@/components/lamma/applications/StatusBadge';
import { MatchScoreRing } from '@/components/lamma/matching/MatchScoreRing';
import { computeMatchScore } from '@/lib/matching/engine';
import { gatherings } from '@/data/gatherings';
import { localized } from '@/lib/use-localized';
import { CheckCircle2, LogIn, Clock } from 'lucide-react';
import { formatRelative } from '@/lib/format';

interface Props {
  gathering: typeof gatherings[number];
}

export function ApplicationCTA({ gathering }: Props) {
  const t = useTranslations('gatherings');
  const tMatching = useTranslations('matching');
  const { user, isAuthenticated, hasHydrated } = useAuthStore();
  const applications = useApplicationsStore();
  const router = useRouter();
  
  const existingApp = user
    ? applications.getByUser(user.id).find(a => a.gatheringSlug === gathering.slug)
    : undefined;
  
  const isClosed = gathering.status === 'APPLICATIONS_CLOSED' || 
    (gathering.applicationsCloseAt && new Date(gathering.applicationsCloseAt) < new Date());
  
  const matchScore = hasHydrated && isAuthenticated && user
    ? computeMatchScore(user, gathering)
    : null;
  
  const handleApply = () => {
    if (!isAuthenticated) {
      window.dispatchEvent(new CustomEvent('lamma:show-login'));
      return;
    }
    router.push(`/gatherings/apply/${gathering.slug}`);
  };
  
  return (
    <Card className="p-6 bg-paper border-clay/20 sticky top-24">
      <div className="flex items-center justify-between mb-4">
        <div>
          <p className="text-sm text-stone">{t('price')}</p>
          <p className="text-3xl font-display tabular">
            {gathering.isFree ? t('free') : `${gathering.priceKwd.toFixed(3)} KWD`}
          </p>
        </div>
        <StatusBadge status={gathering.status} />
      </div>
      
      {/* Match prediction (for logged-in users only) */}
      {hasHydrated && isAuthenticated && matchScore !== null && !existingApp && !isClosed && (
        <div className="mb-4 p-4 bg-sand rounded-lg">
          <p className="text-xs text-stone mb-2 text-center">{tMatching('yourMatchScore')}</p>
          <div className="flex items-center justify-center my-2">
            <MatchScoreRing score={matchScore} size="md" showLabel={false} />
          </div>
          <p className="text-xs text-stone text-center">{tMatching('matchScoreHint')}</p>
        </div>
      )}
      
      {existingApp ? (
        <div className="space-y-3">
          <div className="bg-success/10 border border-success/20 rounded-lg p-3 flex items-center gap-2">
            <CheckCircle2 className="size-5 text-success flex-shrink-0" />
            <div>
              <p className="text-sm font-medium text-success">{t('alreadyApplied')}</p>
              <p className="text-xs text-stone">{t('applicationId')}: {existingApp.id.slice(-8)}</p>
            </div>
          </div>
          <div className="flex items-center justify-between">
            <span className="text-sm text-stone">{t('currentStatus')}</span>
            <StatusBadge status={existingApp.status} />
          </div>
          <Button variant="outline" className="w-full" onClick={() => router.push('/dashboard/applications')}>
            {t('viewInDashboard')}
          </Button>
        </div>
      ) : isClosed ? (
        <div className="space-y-3">
          <div className="bg-stone/10 border border-stone/20 rounded-lg p-3">
            <p className="text-sm text-stone">{t('applicationsClosed')}</p>
          </div>
          <Button disabled className="w-full bg-stone">{t('closed')}</Button>
        </div>
      ) : (
        <>
          <Button size="lg" className="w-full bg-clay hover:bg-clay/90 text-paper" onClick={handleApply}>
            {isAuthenticated ? t('applyCta') : (
              <span className="flex items-center gap-2">
                <LogIn className="size-4" />
                {t('loginToApply')}
              </span>
            )}
          </Button>
          <p className="text-xs text-stone text-center mt-3">{t('applyNote')}</p>
          {gathering.applicationsCloseAt && (
            <p className="text-xs text-saffron text-center mt-2 flex items-center justify-center gap-1">
              <Clock className="size-3" />
              {t('applicationsCloseIn', { time: formatRelative(gathering.applicationsCloseAt, 'ar') })}
            </p>
          )}
        </>
      )}
    </Card>
  );
}
```

### 2. CapacityCard (تحقّق + إكمال)

في `src/components/lamma/gatherings/detail/CapacityCard.tsx`:

```tsx
import { Card } from '@/components/ui/card';
import { Avatar, AvatarImage, AvatarFallback } from '@/components/ui/avatar';
import { useTranslations } from 'next-intl';
import { Users, Clock, ListPlus } from 'lucide-react';
import { localized } from '@/lib/use-localized';

interface Props {
  capacityMax: number;
  capacityMin: number;
  approvedCount: number;
  pendingCount: number;
  waitlistedCount: number;
  approvedAttendees: Array<{ id: string; name: { ar: string; en: string }; avatarUrl?: string }>;
}

export function CapacityCard({ capacityMax, capacityMin, approvedCount, pendingCount, waitlistedCount, approvedAttendees }: Props) {
  const t = useTranslations('gatherings');
  const fillPercent = Math.round((approvedCount / capacityMax) * 100);
  const isFull = approvedCount >= capacityMax;
  
  return (
    <Card className="p-6">
      <div className="flex items-center gap-2 mb-4">
        <Users className="size-5 text-teal" />
        <h3 className="font-display text-lg">{t('capacity')}</h3>
      </div>
      
      <div className="space-y-3">
        {/* Approved */}
        <div>
          <div className="flex justify-between text-sm mb-1">
            <span className="flex items-center gap-1.5">
              <span className="size-2 rounded-full bg-success" />
              {t('approved')}
            </span>
            <span className="tabular">{approvedCount}/{capacityMax}</span>
          </div>
          <div className="h-2 bg-sand rounded-full overflow-hidden">
            <div className="h-full bg-success transition-all duration-500" style={{ width: `${fillPercent}%` }} />
          </div>
        </div>
        
        {/* Pending */}
        {pendingCount > 0 && (
          <div>
            <div className="flex justify-between text-sm mb-1">
              <span className="flex items-center gap-1.5">
                <Clock className="size-3 text-saffron" />
                {t('pending')}
              </span>
              <span className="tabular text-saffron">{pendingCount}</span>
            </div>
            <div className="h-1.5 bg-sand rounded-full overflow-hidden">
              <div className="h-full bg-saffron/60" style={{ width: `${Math.min(100, (pendingCount / capacityMax) * 100)}%` }} />
            </div>
          </div>
        )}
        
        {/* Waitlisted */}
        {waitlistedCount > 0 && (
          <div>
            <div className="flex justify-between text-sm mb-1">
              <span className="flex items-center gap-1.5">
                <ListPlus className="size-3 text-stone" />
                {t('waitlisted')}
              </span>
              <span className="tabular text-stone">{waitlistedCount}</span>
            </div>
          </div>
        )}
        
        {isFull && (
          <p className="text-xs text-saffron mt-2">{t('fullyBooked')}</p>
        )}
      </div>
      
      {/* Avatar grid */}
      {approvedAttendees.length > 0 && (
        <div className="mt-5 pt-4 border-t border-stone/10">
          <p className="text-xs text-stone mb-3">{t('whosComing')}</p>
          <div className="flex flex-wrap gap-2">
            {approvedAttendees.slice(0, 8).map(attendee => (
              <Avatar key={attendee.id} className="size-8 ring-2 ring-paper">
                {attendee.avatarUrl ? <AvatarImage src={attendee.avatarUrl} alt="" /> : null}
                <AvatarFallback className="text-xs">{localized(attendee.name).charAt(0)}</AvatarFallback>
              </Avatar>
            ))}
            {approvedCount > 8 && (
              <div className="size-8 rounded-full bg-sand flex items-center justify-center text-xs text-stone">
                +{approvedCount - 8}
              </div>
            )}
          </div>
        </div>
      )}
    </Card>
  );
}
```

```tsx
'use client';

import { useNotificationsStore } from '@/lib/notifications-store';
import { Bell } from 'lucide-react';
import { Button } from '@/components/ui/button';

  const { notifications, unreadCount, markAllRead, markRead } = useNotificationsStore();

    <DropdownMenu>
      <DropdownMenuTrigger asChild>

          )}

      </DropdownMenuTrigger>
      <DropdownMenuContent align="end" className="w-80">
        <div className="flex items-center justify-between p-2">
          <span className="text-sm font-medium">{t('title')}</span>
          {unreadCount > 0 && (
            <button onClick={markAllRead} className="text-xs text-clay hover:underline">
              {t('markAllRead')}
            </button>
          )}
        </div>
        <DropdownMenuSeparator />
        {notifications.length === 0 ? (
          <div className="p-4 text-center text-sm text-stone">{t('empty')}</div>
        ) : (
          notifications.slice(0, 5).map(notif => (
            <DropdownMenuItem
              key={notif.id}
              onClick={() => markRead(notif.id)}
              className="flex flex-col items-start p-3 cursor-pointer"
            >
              <div className="flex items-center gap-2 w-full">
                {!notif.isRead && <span className="size-2 bg-clay rounded-full flex-shrink-0" />}
                <span className="text-sm font-medium flex-1">{localized(notif.title)}</span>
                <span className="text-xs text-stone">{formatRelative(notif.createdAt, 'ar')}</span>
              </div>
              <p className="text-xs text-stone mt-1">{localized(notif.body)}</p>
            </DropdownMenuItem>
          ))
        )}
      </DropdownMenuContent>
    </DropdownMenu>
  );
}
```

```typescript
  'هذه اللمة مختلفة فعلاً',
  'صاحب التجمع شخص مبدع',
];

const MOCK_MOMENTS = [
  { type: 'highlight' as const, content: 'لحظة الصمت الجماعي عند قراءة القصيدة' },
  { type: 'quote' as const, content: 'الكويت ليست مكاناً، الكويت حكاية' },
  { type: 'highlight' as const, content: 'النقاش حول ذاكرة المكان كان عميقاً' },
];

function seedMockActivity(slug: string) {
  const state = getGathering(slug);
  
  // Add 2 mock attendees
  MOCK_BOTS.slice(0, 2).forEach((bot, i) => {
    state.attendees.set(bot.userId, {
      socketId: `bot-${i}`,
      userId: bot.userId,
      userName: bot.userName,
      avatarUrl: bot.avatarUrl,
      joinedAt: Date.now() - (i + 1) * 60000,
      isHost: false,
    });
  });
  
  // Add 2-3 mock messages
  for (let i = 0; i < 3; i++) {
    const bot = MOCK_BOTS[i % MOCK_BOTS.length];
    state.messages.push({
      id: `mock-msg-${i}-${Date.now()}`,
      gatheringSlug: slug,
      userId: bot.userId,
      userName: bot.userName,
      avatarUrl: bot.avatarUrl,
      content: MOCK_MESSAGES[i % MOCK_MESSAGES.length],
      timestamp: Date.now() - (3 - i) * 120000,
      isSystem: false,
    });
  }
  
  // Add 1 mock moment
  const mockMoment = MOCK_MOMENTS[0];
  state.moments.push({
    id: `mock-moment-${Date.now()}`,
    gatheringSlug: slug,
    userId: MOCK_BOTS[0].userId,
    userName: MOCK_BOTS[0].userName,
    type: mockMoment.type,
    content: mockMoment.content,
    hearts: Math.floor(Math.random() * 5) + 1,
    timestamp: Date.now() - 300000,
  });
}

// ============================================
// Bot activity simulator
// ============================================

function startBotSimulation(slug: string) {
  const state = getGathering(slug);
  
  // Every 30-90 seconds, post a mock message
  const interval = setInterval(() => {
    if (state.attendees.size < 3) return;
    
    const bot = MOCK_BOTS[Math.floor(Math.random() * MOCK_BOTS.length)];
    const msg = MOCK_MESSAGES[Math.floor(Math.random() * MOCK_MESSAGES.length)];
    
    const newMsg: LiveMessage = {
      id: `bot-msg-${Date.now()}`,
      gatheringSlug: slug,
      userId: bot.userId,
      userName: bot.userName,
      avatarUrl: bot.avatarUrl,
      content: msg,
      timestamp: Date.now(),
      isSystem: false,
    };
    
    state.messages.push(newMsg);
    io.to(`gathering:${slug}`).emit('message:new', newMsg);
  }, 45000 + Math.random() * 45000);
  
  return interval;
}

io.on('connection', (socket) => {
  console.log(`[Live] Connected: ${socket.id}`);
  
  socket.on('gathering:join', (payload: {
    gatheringSlug: string;
    userId: string;
    userName: string;
    avatarUrl?: string;
    isHost: boolean;
  }) => {
    const { gatheringSlug, userId, userName, avatarUrl, isHost } = payload;
    
    // Seed mock activity on first join
    const state = getGathering(gatheringSlug);
    if (state.attendees.size === 0 && state.messages.length === 0) {
      seedMockActivity(gatheringSlug);
      startBotSimulation(gatheringSlug);
    }
    
    // Add attendee
    state.attendees.set(userId, {
      socketId: socket.id,
      userId,
      userName,
      avatarUrl,
      joinedAt: Date.now(),
      isHost,
    });
    
    // Join socket room
    socket.join(`gathering:${gatheringSlug}`);
    socket.data.gatheringSlug = gatheringSlug;
    socket.data.userId = userId;
    
    // System message
    const sysMsg: LiveMessage = {
      id: `sys-${Date.now()}`,
      gatheringSlug,
      userId: 'system',
      userName: 'النظام',
      content: `${userName} انضم لللمة`,
      timestamp: Date.now(),
      isSystem: true,
    };
    state.messages.push(sysMsg);
    
    // Send current state to the new joiner
    socket.emit('gathering:state', {
      attendees: Array.from(state.attendees.values()),
      messages: state.messages.slice(-50), // last 50
      moments: state.moments,
    });
    
    // Notify others
    socket.to(`gathering:${gatheringSlug}`).emit('message:new', sysMsg);
    socket.to(`gathering:${gatheringSlug}`).emit('attendee:join', state.attendees.get(userId));
  });
  
  socket.on('message:send', (payload: { content: string }) => {
    const { gatheringSlug, userId } = socket.data;
    if (!gatheringSlug || !userId) return;
    
    const state = getGathering(gatheringSlug);
    const attendee = state.attendees.get(userId);
    if (!attendee) return;
    
    const msg: LiveMessage = {
      id: `msg-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`,
      gatheringSlug,
      userId,
      userName: attendee.userName,
      avatarUrl: attendee.avatarUrl,
      content: payload.content.slice(0, 500), // limit
      timestamp: Date.now(),
      isSystem: false,
    };
    
    state.messages.push(msg);
    io.to(`gathering:${gatheringSlug}`).emit('message:new', msg);
  });
  
  socket.on('moment:create', (payload: { type: 'highlight' | 'quote'; content: string }) => {
    const { gatheringSlug, userId } = socket.data;
    if (!gatheringSlug || !userId) return;
    
    const state = getGathering(gatheringSlug);
    const attendee = state.attendees.get(userId);
    if (!attendee) return;
    
    const moment: LiveMoment = {
      id: `moment-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`,
      gatheringSlug,
      userId,
      userName: attendee.userName,
      type: payload.type,
      content: payload.content.slice(0, 200),
      hearts: 0,
      timestamp: Date.now(),
    };
    
    state.moments.unshift(moment); // add to top
    if (state.moments.length > 20) state.moments.pop();
    
    io.to(`gathering:${gatheringSlug}`).emit('moment:new', moment);
  });
  
  socket.on('moment:heart', (payload: { momentId: string }) => {
    const { gatheringSlug } = socket.data;
    if (!gatheringSlug) return;
    
    const state = getGathering(gatheringSlug);
    const moment = state.moments.find(m => m.id === payload.momentId);
    if (moment) {
      moment.hearts++;
      io.to(`gathering:${gatheringSlug}`).emit('moment:hearted', { momentId: moment.id, hearts: moment.hearts });
    }
  });
  
  socket.on('typing:start', () => {
    const { gatheringSlug, userId } = socket.data;
    if (!gatheringSlug || !userId) return;
    const state = getGathering(gatheringSlug);
    const attendee = state.attendees.get(userId);
    if (attendee) {
      socket.to(`gathering:${gatheringSlug}`).emit('typing:start', { userId, userName: attendee.userName });
    }
  });
  
  socket.on('typing:stop', () => {
    const { gatheringSlug, userId } = socket.data;
    if (!gatheringSlug || !userId) return;
    socket.to(`gathering:${gatheringSlug}`).emit('typing:stop', { userId });
  });
  
  socket.on('disconnect', () => {
    const { gatheringSlug, userId } = socket.data;
    if (!gatheringSlug || !userId) return;
    
    const state = getGathering(gatheringSlug);
    const attendee = state.attendees.get(userId);
    state.attendees.delete(userId);
    
    if (attendee) {
      const sysMsg: LiveMessage = {
        id: `sys-${Date.now()}`,
        gatheringSlug,
        userId: 'system',
        userName: 'النظام',
        content: `${attendee.userName} غادر اللمة`,
        timestamp: Date.now(),
        isSystem: true,
      };
      state.messages.push(sysMsg);
      socket.to(`gathering:${gatheringSlug}`).emit('message:new', sysMsg);
      socket.to(`gathering:${gatheringSlug}`).emit('attendee:leave', { userId });
    }
  });
});

httpServer.listen(PORT, () => {
  console.log(`[Lamma Live Companion] Running on port ${PORT}`);
});
```

### LiveCompanion component

`src/components/lamma/live/LiveCompanion.tsx`:

```tsx
'use client';

import { useEffect, useState, useRef } from 'react';
import { io, Socket } from 'socket.io-client';
import { useAuthStore } from '@/lib/auth-store';
import { useTranslations } from 'next-intl';
import { Card } from '@/components/ui/card';
import { Button } from '@/components/ui/button';
import { Input } from '@/components/ui/input';
import { Avatar, AvatarImage, AvatarFallback } from '@/components/ui/avatar';
import { Send, Heart, Sparkles, Users, Clock, MessageSquare, LogOut, Quote, Star } from 'lucide-react';
import { PrayerTimeBadge } from '@/components/lamma/PrayerTimeBadge';
import { localized } from '@/lib/use-localized';
import { cn } from '@/lib/utils';
import Link from 'next/link';

interface Attendee {
  socketId: string;
  userId: string;
  userName: string;
  avatarUrl?: string;
  joinedAt: number;
  isHost: boolean;
}

interface Message {
  id: string;
  userId: string;
  userName: string;
  avatarUrl?: string;
  content: string;
  timestamp: number;
  isSystem: boolean;
}

interface Moment {
  id: string;
  userId: string;
  userName: string;
  type: 'highlight' | 'quote' | 'photo-placeholder';
  content: string;
  hearts: number;
  timestamp: number;
}

interface Props {
  gathering: any; // Gathering type
}

export function LiveCompanion({ gathering }: Props) {
  const t = useTranslations('live');
  const { user } = useAuthStore();
  
  const [socket, setSocket] = useState<Socket | null>(null);
  const [connected, setConnected] = useState(false);
  const [attendees, setAttendees] = useState<Attendee[]>([]);
  const [messages, setMessages] = useState<Message[]>([]);
  const [moments, setMoments] = useState<Moment[]>([]);
  const [input, setInput] = useState('');
  const [typingUsers, setTypingUsers] = useState<Map<string, string>>(new Map());
  const [elapsedMs, setElapsedMs] = useState(0);
  const [activeTab, setActiveTab] = useState<'chat' | 'moments' | 'attendees'>('chat');
  
  const messagesEndRef = useRef<HTMLDivElement>(null);
  const typingTimeoutRef = useRef<NodeJS.Timeout | null>(null);
  
  // Connect to WebSocket
  useEffect(() => {
    if (!user) return;
    
    // مهم: استخدم path '/' و XTransformPort في query (per fullstack-dev skill)
    const newSocket = io('/', {
      transports: ['websocket'],
      query: { XTransformPort: '3003' },
    });
    
    setSocket(newSocket);
    
    newSocket.on('connect', () => {
      setConnected(true);
      newSocket.emit('gathering:join', {
        gatheringSlug: gathering.slug,
        userId: user.id,
        userName: user.nameLocalized.ar,
        avatarUrl: user.avatarUrl,
        isHost: user.membershipTier === 'HOST',
      });
    });
    
    newSocket.on('disconnect', () => setConnected(false));
    
    newSocket.on('gathering:state', (state: {
      attendees: Attendee[];
      messages: Message[];
      moments: Moment[];
    }) => {
      setAttendees(state.attendees);
      setMessages(state.messages);
      setMoments(state.moments);
    });
    
    newSocket.on('message:new', (msg: Message) => {
      setMessages(prev => [...prev, msg]);
    });
    
    newSocket.on('attendee:join', (attendee: Attendee) => {
      setAttendees(prev => {
        if (prev.find(a => a.userId === attendee.userId)) return prev;
        return [...prev, attendee];
      });
    });
    
    newSocket.on('attendee:leave', ({ userId }: { userId: string }) => {
      setAttendees(prev => prev.filter(a => a.userId !== userId));
    });
    
    newSocket.on('moment:new', (moment: Moment) => {
      setMoments(prev => [moment, ...prev]);
    });
    
    newSocket.on('moment:hearted', ({ momentId, hearts }: { momentId: string; hearts: number }) => {
      setMoments(prev => prev.map(m => m.id === momentId ? { ...m, hearts } : m));
    });
    
    newSocket.on('typing:start', ({ userId, userName }: { userId: string; userName: string }) => {
      setTypingUsers(prev => new Map(prev).set(userId, userName));
    });
    
    newSocket.on('typing:stop', ({ userId }: { userId: string }) => {
      setTypingUsers(prev => {
        const next = new Map(prev);
        next.delete(userId);
        return next;
      });
    });
    
    return () => {
      newSocket.close();
    };
  }, [user, gathering.slug]);
  
  // Scroll to bottom on new message
  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: 'smooth' });
  }, [messages]);
  
  // Elapsed time counter
  useEffect(() => {
    const startTime = new Date(gathering.startDate).getTime();
    const interval = setInterval(() => {
      setElapsedMs(Date.now() - startTime);
    }, 1000);
    return () => clearInterval(interval);
  }, [gathering.startDate]);
  
  const sendMessage = () => {
    if (!input.trim() || !socket) return;
    socket.emit('message:send', { content: input.trim() });
    setInput('');
    socket.emit('typing:stop');
  };
  
  const handleInputChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    setInput(e.target.value);
    if (!socket) return;
    
    socket.emit('typing:start');
    
    if (typingTimeoutRef.current) clearTimeout(typingTimeoutRef.current);
    typingTimeoutRef.current = setTimeout(() => {
      socket.emit('typing:stop');
    }, 2000);
  };
  
  const createMoment = (type: 'highlight' | 'quote') => {
    const content = type === 'highlight' 
      ? prompt(t('momentHighlightPrompt'))
      : prompt(t('momentQuotePrompt'));
    
    if (!content || !socket) return;
    socket.emit('moment:create', { type, content });
  };
  
  const heartMoment = (momentId: string) => {
    if (socket) socket.emit('moment:heart', { momentId });
  };
  
  const formatElapsed = (ms: number) => {
    const totalSec = Math.floor(ms / 1000);
    const h = Math.floor(totalSec / 3600);
    const m = Math.floor((totalSec % 3600) / 60);
    const s = totalSec % 60;
    return `${String(h).padStart(2, '0')}:${String(m).padStart(2, '0')}:${String(s).padStart(2, '0')}`;
  };
  
  return (
    <div className="container py-6 max-w-6xl">
      {/* Header */}
      <div className="mb-6 flex flex-wrap items-center justify-between gap-4">
        <div>
          <Link href={`/gatherings/${gathering.slug}`} className="text-xs text-stone hover:text-clay mb-1 inline-block">
            ← {t('backToGathering')}
          </Link>
          <h1 className="font-display text-2xl md:text-3xl">{localized(gathering.title)}</h1>
          <div className="flex flex-wrap items-center gap-3 mt-2 text-sm text-stone">
            <span className={cn('flex items-center gap-1.5', connected ? 'text-success' : 'text-stone')}>
              <span className={cn('size-2 rounded-full', connected ? 'bg-success animate-pulse' : 'bg-stone')} />
              {connected ? t('live') : t('connecting')}
            </span>
            <span className="flex items-center gap-1.5">
              <Clock className="size-3.5" />
              <span className="tabular">{formatElapsed(elapsedMs)}</span>
            </span>
            <span className="flex items-center gap-1.5">
              <Users className="size-3.5" />
              {attendees.length} {t('hereNow')}
            </span>
          </div>
        </div>
        <PrayerTimeBadge 
          startDate={new Date(gathering.startDate)} 
          endDate={new Date(gathering.endDate)} 
          variant="compact" 
        />
      </div>
      
      {/* Tabs (native buttons) */}
      <div className="inline-flex h-10 items-center justify-center rounded-lg bg-sand p-1 mb-4">
        {[
          { id: 'chat', label: t('chat'), icon: MessageSquare },
          { id: 'moments', label: t('moments'), icon: Star },
          { id: 'attendees', label: t('attendees'), icon: Users },
        ].map(tab => {
          const Icon = tab.icon;
          return (
            <button
              key={tab.id}
              onClick={() => setActiveTab(tab.id as any)}
              className={cn(
                'inline-flex items-center gap-1.5 px-3 py-1.5 text-sm rounded-md transition-all',
                activeTab === tab.id ? 'bg-paper text-ink shadow-sm' : 'text-stone hover:text-ink'
              )}
            >
              <Icon className="size-3.5" />
              {tab.label}
            </button>
          );
        })}
      </div>
      
      <div className="grid lg:grid-cols-[1fr_300px] gap-4">
        {/* Main content */}
        <div>
          {activeTab === 'chat' && (
            <Card className="flex flex-col h-[60vh]">
              {/* Messages */}
              <div className="flex-1 overflow-y-auto p-4 space-y-3">
                {messages.map(msg => (
                  <div key={msg.id} className={cn('flex gap-2', msg.isSystem && 'justify-center')}>
                    {msg.isSystem ? (
                      <p className="text-xs text-stone italic">{msg.content}</p>
                    ) : (
                      <>
                        <Avatar className="size-8 flex-shrink-0">
                          {msg.avatarUrl && <AvatarImage src={msg.avatarUrl} alt="" />}
                          <AvatarFallback className="text-xs">{msg.userName.charAt(0)}</AvatarFallback>
                        </Avatar>
                        <div className="flex-1 min-w-0">
                          <div className="flex items-baseline gap-2">
                            <span className="text-sm font-medium">{msg.userName}</span>
                            <span className="text-xs text-stone tabular">
                              {new Date(msg.timestamp).toLocaleTimeString('ar', { hour: '2-digit', minute: '2-digit' })}
                            </span>
                          </div>
                          <p className="text-sm text-ink/80 break-words">{msg.content}</p>
                        </div>
                      </>
                    )}
                  </div>
                ))}
                <div ref={messagesEndRef} />
              </div>
              
              {/* Typing indicator */}
              {typingUsers.size > 0 && (
                <div className="px-4 pb-2 text-xs text-stone italic">
                  {Array.from(typingUsers.values()).join('، ')} {t('typing')}
                </div>
              )}
              
              {/* Input */}
              <div className="border-t border-stone/20 p-3 flex gap-2">
                <Input
                  value={input}
                  onChange={handleInputChange}
                  placeholder={t('messagePlaceholder')}
                  onKeyDown={(e) => e.key === 'Enter' && sendMessage()}
                  disabled={!connected}
                />
                <Button onClick={sendMessage} disabled={!input.trim() || !connected} className="bg-clay hover:bg-clay/90 text-paper">
                  <Send className="size-4" />
                </Button>
              </div>
            </Card>
          )}
          
          {activeTab === 'moments' && (
            <div className="space-y-3">
              <div className="flex gap-2">
                <Button variant="outline" size="sm" onClick={() => createMoment('highlight')}>
                  <Star className="size-3.5 me-1.5" />
                  {t('addHighlight')}
                </Button>
                <Button variant="outline" size="sm" onClick={() => createMoment('quote')}>
                  <Quote className="size-3.5 me-1.5" />
                  {t('addQuote')}
                </Button>
              </div>
              
              {moments.length === 0 ? (
                <Card className="p-8 text-center text-stone">
                  <Star className="size-8 mx-auto mb-2 opacity-30" />
                  <p className="text-sm">{t('noMomentsYet')}</p>
                </Card>
              ) : (
                moments.map(moment => (
                  <Card key={moment.id} className="p-4">
                    <div className="flex items-start gap-3">
                      <div className={cn(
                        'size-10 rounded-full flex items-center justify-center flex-shrink-0',
                        moment.type === 'highlight' ? 'bg-saffron/15 text-saffron' : 'bg-clay/15 text-clay'
                      )}>
                        {moment.type === 'highlight' ? <Star className="size-5" /> : <Quote className="size-5" />}
                      </div>
                      <div className="flex-1">
                        <p className="text-sm font-medium mb-1">{moment.userName}</p>
                        <p className={cn(
                          'text-base',
                          moment.type === 'quote' && 'italic font-display'
                        )}>
                          {moment.type === 'quote' ? `«${moment.content}»` : moment.content}
                        </p>
                        <div className="flex items-center justify-between mt-2">
                          <span className="text-xs text-stone tabular">
                            {new Date(moment.timestamp).toLocaleTimeString('ar', { hour: '2-digit', minute: '2-digit' })}
                          </span>
                          <button
                            onClick={() => heartMoment(moment.id)}
                            className="flex items-center gap-1 text-xs text-stone hover:text-clay transition-colors"
                          >
                            <Heart className="size-3.5" />
                            <span className="tabular">{moment.hearts}</span>
                          </button>
                        </div>
                      </div>
                    </div>
                  </Card>
                ))
              )}
            </div>
          )}
          
          {activeTab === 'attendees' && (
            <Card className="p-4">
              <h3 className="font-display text-lg mb-3">{t('whosHere')}</h3>
              <div className="space-y-2">
                {attendees.map(attendee => (
                  <div key={attendee.userId} className="flex items-center gap-3 p-2 rounded-lg hover:bg-sand transition-colors">
                    <div className="relative">
                      <Avatar className="size-10">
                        {attendee.avatarUrl && <AvatarImage src={attendee.avatarUrl} alt="" />}
                        <AvatarFallback>{attendee.userName.charAt(0)}</AvatarFallback>
                      </Avatar>
                      <span className="absolute -bottom-0.5 -end-0.5 size-3 bg-success rounded-full ring-2 ring-paper" />
                    </div>
                    <div className="flex-1">
                      <p className="text-sm font-medium flex items-center gap-1.5">
                        {attendee.userName}
                        {attendee.isHost && (
                          <span className="text-xs px-1.5 py-0.5 bg-clay/15 text-clay rounded">{t('host')}</span>
                        )}
                      </p>
                      <p className="text-xs text-stone">
                        {t('joinedAgo', { time: Math.round((Date.now() - attendee.joinedAt) / 60000) })}
                      </p>
                    </div>
                  </div>
                ))}
              </div>
            </Card>
          )}
        </div>
        
        {/* Sidebar */}
        <div className="space-y-4">
          {/* Attendees preview */}
          <Card className="p-4">
            <div className="flex items-center gap-2 mb-3">
              <Users className="size-4 text-teal" />
              <h3 className="font-display text-sm">{t('attendeesCount', { count: attendees.length })}</h3>
            </div>
            <div className="flex flex-wrap gap-2">
              {attendees.slice(0, 8).map(a => (
                <Avatar key={a.userId} className="size-8 ring-2 ring-paper">
                  {a.avatarUrl && <AvatarImage src={a.avatarUrl} alt="" />}
                  <AvatarFallback className="text-xs">{a.userName.charAt(0)}</AvatarFallback>
                </Avatar>
              ))}
            </div>
          </Card>
          
          {/* Prayer reminder */}
          <Card className="p-4 bg-sand/50">
            <div className="flex items-center gap-2 mb-2">
              <Sparkles className="size-4 text-clay" />
              <h3 className="font-display text-sm">{t('prayerReminder')}</h3>
            </div>
            <p className="text-xs text-stone">{t('prayerReminderHint')}</p>
          </Card>
          
          {/* Quick actions */}
          <Card className="p-4">
            <h3 className="font-display text-sm mb-3">{t('quickActions')}</h3>
            <div className="space-y-2">
              <Link href={`/gatherings/${gathering.slug}`} className="block text-xs text-teal hover:underline">
                ← {t('backToGathering')}
              </Link>
            </div>
          </Card>
        </div>
      </div>
    </div>
  );
}
```

```tsx
// إذا الـ gathering قادمة خلال 24 ساعة أو جارية الآن، اعرض زر "Live"
const now = new Date();

const endTime = new Date(gathering.endDate);
const isLive = now >= startTime && now <= endTime;
const isUpcoming = startTime > now && startTime.getTime() - now.getTime() < 24 * 60 * 60 * 1000;

{(isLive || isUpcoming) && (
  <Button asChild className="bg-success hover:bg-success/90 text-paper">

)}
```

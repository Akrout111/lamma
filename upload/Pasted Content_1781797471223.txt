### Helper: حساب `isHost` و `hostedGatherings`

أنشئ `src/lib/host-helpers.ts`:

```typescript
import { hosts } from '@/data/hosts';
import { gatherings } from '@/data/gatherings';
import type { DemoUser } from '@/data/demo-users';

export function getUserHost(userId: string) {
  return hosts.find(h => h.userId === userId);
}

export function isUserHost(userId: string): boolean {
  return !!getUserHost(userId);
}

export function getHostGatherings(hostHandle: string) {
  return gatherings.filter(g => g.hostHandle === hostHandle);
}

export function getHostUpcomingGatherings(hostHandle: string) {
  const now = new Date();
  return getHostGatherings(hostHandle).filter(g => new Date(g.startDate) > now);
}

export function getHostPastGatherings(hostHandle: string) {
  const now = new Date();
  return getHostGatherings(hostHandle).filter(g => new Date(g.startDate) <= now);
}
```

```tsx
'use client';
```

```tsx
const navItems = [
```

### Store for host-created gatherings

`src/lib/host-gatherings-store.ts` (Zustand + persist):

```typescript
'use client';

import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { z } from 'zod';

const HostGatheringSchema = z.object({
  id: z.string(),
  slug: z.string(),
  hostHandle: z.string(),
  // ... كل الحقول من gatherings.ts schema
  status: z.literal('DRAFT').default('DRAFT'),
  createdAt: z.string(),
});

type HostGathering = z.infer<typeof HostGatheringSchema>;

interface State {
  gatherings: HostGathering[];
  create: (data: Omit<HostGathering, 'id' | 'createdAt' | 'status'>) => HostGathering;
  // لا edit/delete في هذه المرحلة
}

export const useHostGatheringsStore = create<State>()(
  persist(
    (set) => ({
      gatherings: [],
      create: (data) => {
        const newG: HostGathering = {
          ...data,
          id: `host-g-${Date.now()}`,
          status: 'DRAFT',
          createdAt: new Date().toISOString(),
        };
        set(state => ({ gatherings: [...state.gatherings, newG] }));
        return newG;
      },
    }),
    { name: 'lamma-host-gatherings' }
  )
);
```

### Notification store (for badge count)

`src/lib/notifications-store.ts` (Zustand + persist):
```typescript
'use client';

import { create } from 'zustand';
import { persist } from 'zustand/middleware';

interface Notification {
  id: string;
  type: 'new_application' | 'application_cancelled' | 'gathering_published' | 'system';
  title: { ar: string; en: string };
  body: { ar: string; en: string };
  data?: Record<string, unknown>;
  isRead: boolean;
  createdAt: string;
}

interface State {
  notifications: Notification[];
  unreadCount: number;
  add: (n: Omit<Notification, 'id' | 'isRead' | 'createdAt'>) => void;
  markAllRead: () => void;
  markRead: (id: string) => void;
}

export const useNotificationsStore = create<State>()(
  persist(
    (set, get) => ({
      notifications: [
        // Seed: 2-3 notifications وهمية
        {
          id: 'notif-1',
          type: 'new_application',
          title: { ar: 'طلب جديد', en: 'New application' },
          body: { ar: 'سارة قدّمت طلباً على "أمسية شعراء كويتيين"', en: 'Sara applied to "Kuwaiti Poets Evening"' },
          isRead: false,
          createdAt: new Date(Date.now() - 3600000).toISOString(),
        },
        // ...
      ],
      unreadCount: 2,
      add: (n) => set(state => ({
        notifications: [{ ...n, id: `notif-${Date.now()}`, isRead: false, createdAt: new Date().toISOString() }, ...state.notifications],
        unreadCount: state.unreadCount + 1,
      })),
      markAllRead: () => set(state => ({
        notifications: state.notifications.map(n => ({ ...n, isRead: true })),
        unreadCount: 0,
      })),
      markRead: (id) => set(state => ({
        notifications: state.notifications.map(n => n.id === id ? { ...n, isRead: true } : n),
        unreadCount: Math.max(0, state.unreadCount - 1),
      })),
    }),
    { name: 'lamma-notifications' }
  )
);
```

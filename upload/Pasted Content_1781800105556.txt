  marketing: ['brand', 'advertising', 'تسويق', 'علامة', 'إعلان'],
};

// Keyword weights — interests أثقل من bio
const WEIGHTS = {
  interest: 3,
  topic: 2,
  title: 1.5,
  description: 1,
  bio: 1,
  nationality: 0.5,
};

// ============================================
// Tokenization
// ============================================

function tokenize(text: string): string[] {
  if (!text) return [];
  // Lowercase + split by non-alphanumeric (يدعم عربي وإنجليزي)
  const cleaned = text.toLowerCase().replace(/[^\p{L}\p{N}\s]/gu, ' ');
  return cleaned
    .split(/\s+/)
    .filter(token => token.length > 1 && !STOP_WORDS.has(token));
}

function expandWithSynonyms(tokens: string[]): string[] {
  const expanded = new Set(tokens);
  for (const token of tokens) {
    // Find synonyms for this token
    for (const [key, synonyms] of Object.entries(SYNONYMS)) {
      if (token === key || synonyms.includes(token)) {
        expanded.add(key);
        synonyms.forEach(s => expanded.add(s));
      }
    }
  }
  return Array.from(expanded);
}

// ============================================
// Vector Building (Bag of Words with TF-IDF-like weighting)
// ============================================

export interface ProfileVector {
  tokens: Map<string, number>; // token → weight
  magnitude: number;
}

function buildVector(weightedTokens: Array<{ token: string; weight: number }>): ProfileVector {
  const tokens = new Map<string, number>();
  
  for (const { token, weight } of weightedTokens) {
    const expanded = expandWithSynonyms([token]);
    for (const t of expanded) {
      tokens.set(t, (tokens.get(t) ?? 0) + weight);
    }
  }
  
  // Calculate magnitude
  let magnitude = 0;
  for (const weight of tokens.values()) {
    magnitude += weight * weight;
  }
  magnitude = Math.sqrt(magnitude);
  
  return { tokens, magnitude };
}

// ============================================
// Profile Builders
// ============================================

import { demoUsers, type DemoUser } from '@/data/demo-users';
import { gatherings, type Gathering } from '@/data/gatherings';
import { hosts } from '@/data/hosts';
import { topics } from '@/data/topics';

export function buildUserVector(user: DemoUser): ProfileVector {
  const weightedTokens: Array<{ token: string; weight: number }> = [];
  
  // Interests (highest weight)
  for (const interest of user.interests) {
    weightedTokens.push({ token: interest, weight: WEIGHTS.interest });
  }
  
  // Bio
  const bioArTokens = tokenize(user.bioLocalized.ar);
  const bioEnTokens = tokenize(user.bioLocalized.en);
  for (const token of [...bioArTokens, ...bioEnTokens]) {
    weightedTokens.push({ token, weight: WEIGHTS.bio });
  }
  
  // Nationality
  if (user.nationality) {
    weightedTokens.push({ token: user.nationality.toLowerCase(), weight: WEIGHTS.nationality });
  }
  
  return buildVector(weightedTokens);
}

export function buildGatheringVector(gathering: Gathering): ProfileVector {
  const weightedTokens: Array<{ token: string; weight: number }> = [];
  
  // Topic slug (high weight)
  weightedTokens.push({ token: gathering.topicSlug, weight: WEIGHTS.topic });
  
  // Topic name (find topic, get its name in both languages)
  const topic = topics.find(t => t.slug === gathering.topicSlug);
  if (topic) {
    for (const t of tokenize(topic.name.ar)) weightedTokens.push({ token: t, weight: WEIGHTS.topic });
    for (const t of tokenize(topic.name.en)) weightedTokens.push({ token: t, weight: WEIGHTS.topic });
  }
  
  // Title
  for (const t of tokenize(gathering.title.ar)) weightedTokens.push({ token: t, weight: WEIGHTS.title });
  for (const t of tokenize(gathering.title.en)) weightedTokens.push({ token: t, weight: WEIGHTS.title });
  
  // Description
  for (const t of tokenize(gathering.description.ar)) weightedTokens.push({ token: t, weight: WEIGHTS.description });
  for (const t of tokenize(gathering.description.en)) weightedTokens.push({ token: t, weight: WEIGHTS.description });
  
  return buildVector(weightedTokens);
}

// ============================================
// Similarity Computation
// ============================================

export function cosineSimilarity(a: ProfileVector, b: ProfileVector): number {
  if (a.magnitude === 0 || b.magnitude === 0) return 0;
  
  let dotProduct = 0;
  // Iterate over the smaller vector for efficiency
  const [smaller, larger] = a.tokens.size < b.tokens.size ? [a, b] : [b, a];
  
  for (const [token, weight] of smaller.tokens) {
    const otherWeight = larger.tokens.get(token);
    if (otherWeight) {
      dotProduct += weight * otherWeight;
    }
  }
  
  return dotProduct / (a.magnitude * b.magnitude);
}

// ============================================
// Public API
// ============================================

/**
 * Compute match score between a user and a gathering.
 * Returns 0-100 (rounded).
 * 
 * Factors:
 * 1. Cosine similarity between user vector and gathering vector
 * 2. Topic interest alignment (user.interests vs gathering.topicSlug)
 * 3. Demographic fit (format compatibility)
 */
export function computeMatchScore(user: DemoUser, gathering: Gathering): number {
  const userVec = buildUserVector(user);
  const gatheringVec = buildGatheringVector(gathering);
  
  // 1. Vector similarity (0-1)
  const vectorSim = cosineSimilarity(userVec, gatheringVec);
  
  // 2. Topic interest alignment
  const topicObj = topics.find(t => t.slug === gathering.topicSlug);
  let topicAlignment = 0;
  if (topicObj) {
    const topicKeywords = [
      gathering.topicSlug,
      ...tokenize(topicObj.name.ar),
      ...tokenize(topicObj.name.en),
    ];
    const matchingInterests = user.interests.filter(interest =>
      topicKeywords.some(kw => 
        kw === interest || 
        (SYNONYMS[interest] && SYNONYMS[interest].includes(kw))
      )
    );
    topicAlignment = matchingInterests.length / Math.max(user.interests.length, 1);
  }
  
  // 3. Format compatibility (basic)
  let formatFit = 0.5;
  if (gathering.format === 'MEN_ONLY' && user.gender === 'male') formatFit = 1;
  if (gathering.format === 'WOMEN_ONLY' && user.gender === 'female') formatFit = 1;
  if (gathering.format === 'MIXED' || gathering.format === 'FAMILY') formatFit = 0.8;
  
  // Weighted combination
  const rawScore = (vectorSim * 0.5) + (topicAlignment * 0.35) + (formatFit * 0.15);
  
  // Normalize 0-1 → 50-100 (avoid very low scores that feel punishing)
  const normalized = 0.5 + (rawScore * 0.5);
  
  return Math.round(Math.min(100, Math.max(50, normalized * 100)));
}

/**
 * Compute similarity between two users (for "people you might know").
 * Returns 0-1.
 */
export function computeUserSimilarity(userA: DemoUser, userB: DemoUser): number {
  const vecA = buildUserVector(userA);
  const vecB = buildUserVector(userB);
  return cosineSimilarity(vecA, vecB);
}

/**
 * Find shared interests between two users.
 */
export function findSharedInterests(userA: DemoUser, userB: DemoUser): string[] {
  const setA = new Set(userA.interests);
  return userB.interests.filter(i => setA.has(i));
}

/**
 * Recommend gatherings for a user.
 * Returns top N gatherings sorted by match score.
 */
export function recommendGatheringsForUser(
  user: DemoUser,
  options: { limit?: number; excludePast?: boolean; excludeApplied?: string[] } = {}
): Array<{ gathering: Gathering; score: number }> {
  const { limit = 3, excludePast = true, excludeApplied = [] } = options;
  const now = new Date();
  
  const candidates = gatherings
    .filter(g => {
      if (excludePast && new Date(g.startDate) < now) return false;
      if (excludeApplied.includes(g.slug)) return false;
      if (g.status === 'CANCELLED' || g.status === 'COMPLETED') return false;
      // Format filter
      if (g.format === 'MEN_ONLY' && user.gender !== 'male') return false;
      if (g.format === 'WOMEN_ONLY' && user.gender !== 'female') return false;
      return true;
    })
    .map(g => ({ gathering: g, score: computeMatchScore(user, g) }))
    .sort((a, b) => b.score - a.score)
    .slice(0, limit);
  
  return candidates;
}

/**
 * Find similar attendees for a gathering (people who would match well).
 * Used in: gathering detail page ("who else might be there").
 */
export function findSimilarAttendees(
  user: DemoUser,
  gatheringSlug: string,
  limit = 3
): Array<{ user: DemoUser; similarity: number; sharedInterests: string[] }> {
  // Find other demo users (excluding self and host)
  const others = demoUsers.filter(u => 
    u.id !== user.id && u.membershipTier !== 'HOST'
  );
  
  // Find existing approved attendees (mock — get from applications-store if available)
  // For now, just return similar users
  return others
    .map(other => ({
      user: other,
      similarity: computeUserSimilarity(user, other),
      sharedInterests: findSharedInterests(user, other),
    }))
    .filter(item => item.similarity > 0.1)
    .sort((a, b) => b.similarity - a.similarity)
    .slice(0, limit);
}

/**
 * Compute merit score for waitlist ranking.
 * Used in: gathering detail when capacity is full.
 * 
 * Factors (per Lamma spec section 8.9):
 * 1. Previous attendance with this host (40%)
 * 2. Membership tier (30%): CURATOR > REGULAR > NEWCOMER
 * 3. Interest match with gathering (30%)
 */
export function computeMeritScore(
  user: DemoUser,
  gathering: Gathering,
  previousAttendanceWithHost: number = 0
): number {
  const tierScores = { NEWCOMER: 0.3, REGULAR: 0.6, CURATOR: 0.9, HOST: 1 };
  const tierScore = tierScores[user.membershipTier] ?? 0.3;
  
  const interestMatch = computeMatchScore(user, gathering) / 100;
  
  // Previous attendance (normalized 0-1, capped at 3)
  const attendanceScore = Math.min(1, previousAttendanceWithHost / 3);
  
  const merit = (attendanceScore * 0.4) + (tierScore * 0.3) + (interestMatch * 0.3);
  return Math.round(merit * 100);
}

'use client';

import { cn } from '@/lib/utils';

interface Props {
  score: number; // 0-100
  size?: 'sm' | 'md' | 'lg';
  showLabel?: boolean;
  label?: string;
}

export function MatchScoreRing({ score, size = 'md', showLabel = true, label }: Props) {
  const dimensions = {
    sm: { box: 48, stroke: 4, font: 'text-xs' },
    md: { box: 80, stroke: 6, font: 'text-base' },
    lg: { box: 120, stroke: 8, font: 'text-2xl' },
  };
  const { box, stroke, font } = dimensions[size];
  const radius = (box - stroke) / 2;
  const circumference = 2 * Math.PI * radius;
  const offset = circumference - (score / 100) * circumference;
  
  const color = score >= 80 ? '#4A7C59' : score >= 65 ? '#E8A93C' : '#8B8478';
  
  return (
    <div className="inline-flex flex-col items-center gap-1">
      <div className="relative" style={{ width: box, height: box }}>
        <svg width={box} height={box} className="-rotate-90">
          <circle
            cx={box / 2}
            cy={box / 2}
            r={radius}
            fill="none"
            stroke="#F4EDE0"
            strokeWidth={stroke}
          />
          <circle
            cx={box / 2}
            cy={box / 2}
            r={radius}
            fill="none"
            stroke={color}
            strokeWidth={stroke}
            strokeDasharray={circumference}
            strokeDashoffset={offset}
            strokeLinecap="round"
            className="transition-all duration-700 ease-out"
          />
        </svg>
        <div className="absolute inset-0 flex items-center justify-center">
          <span className={cn('font-display tabular font-semibold', font)} style={{ color }}>
            {score}
          </span>
        </div>
      </div>
      {showLabel && label && (
        <span className="text-xs text-stone">{label}</span>
      )}
    </div>
  );
}

### "لماذا نوصي بهذه اللمة؟" — Explainable AI
لكل توصية، اعرض 2-3 أسباب:
- "يطابق اهتمامك: {interest}"
- "{count} من أشخاص مشابهين سيحضرون"
- "موضوع يهمك: {topic name}"
- "حجم مناسب ({capacity} شخص)"

أنشئ `src/lib/matching/explain.ts`:

```typescript
import type { DemoUser } from '@/data/demo-users';
import type { Gathering } from '@/data/gatherings';
import { findSharedInterests, findSimilarAttendees } from './engine';
import { topics } from '@/data/topics';

export function explainRecommendation(
  user: DemoUser,
  gathering: Gathering,
  score: number
): string[] {
  const reasons: string[] = [];
  const topic = topics.find(t => t.slug === gathering.topicSlug);
  
  // Topic interest match
  if (topic) {
    const topicKeywords = [
      gathering.topicSlug,
      ...topic.name.ar.split(' '),
      ...topic.name.en.split(' '),
    ];
    const matchingInterests = user.interests.filter(i =>
      topicKeywords.some(kw => kw.toLowerCase().includes(i.toLowerCase()) || i.toLowerCase().includes(kw.toLowerCase()))
    );
    if (matchingInterests.length > 0) {
      reasons.push(`matching-interest:${matchingInterests[0]}`);
    }
  }
  
  // Capacity fit (intimate size)
  if (gathering.capacityMax <= 20) {
    reasons.push('intimate-size');
  }
  
  // Similar attendees
  const similar = findSimilarAttendees(user, gathering.slug, 5);
  if (similar.length > 0) {
    reasons.push(`similar-attendees:${similar.length}`);
  }
  
  // Topic editorial
  if (topic) {
    reasons.push(`topic:${topic.slug}`);
  }
  
  return reasons.slice(0, 3);
}
```

## 🔌 API Routes

### 1. `/api/v1/match-score` — POST

`src/app/api/v1/match-score/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { computeMatchScore } from '@/lib/matching/engine';
import { demoUsers } from '@/data/demo-users';
import { gatherings } from '@/data/gatherings';
import { z } from 'zod';

const BodySchema = z.object({
  userId: z.string(),
  gatheringSlug: z.string(),
});

export async function POST(req: NextRequest) {
  try {
    const body = await req.json();
    const { userId, gatheringSlug } = BodySchema.parse(body);
    
    const user = demoUsers.find(u => u.id === userId);
    const gathering = gatherings.find(g => g.slug === gatheringSlug);
    
    if (!user || !gathering) {
      return NextResponse.json(
        { success: false, error: 'User or gathering not found' },
        { status: 404 }
      );
    }
    
    const score = computeMatchScore(user, gathering);
    
    return NextResponse.json({
      success: true,
      score,
      user: { id: user.id, name: user.nameLocalized },
      gathering: { slug: gathering.slug, title: gathering.title },
    });
  } catch (error) {
    return NextResponse.json(
      { success: false, error: 'Invalid request' },
      { status: 400 }
    );
  }
}
```

### 2. `/api/v1/recommendations` — GET

`src/app/api/v1/recommendations/route.ts`:

```typescript
import { NextRequest, NextResponse } from 'next/server';
import { recommendGatheringsForUser, explainRecommendation } from '@/lib/matching/engine';
import { demoUsers } from '@/data/demo-users';

export async function GET(req: NextRequest) {
  const userId = req.nextUrl.searchParams.get('userId');
  const limit = parseInt(req.nextUrl.searchParams.get('limit') ?? '3');
  
  if (!userId) {
    return NextResponse.json(
      { success: false, error: 'userId required' },
      { status: 400 }
    );
  }
  
  const user = demoUsers.find(u => u.id === userId);
  if (!user) {
    return NextResponse.json(
      { success: false, error: 'User not found' },
      { status: 404 }
    );
  }
  
  const recommendations = recommendGatheringsForUser(user, { limit }).map(({ gathering, score }) => ({
    gathering: {
      slug: gathering.slug,
      title: gathering.title,
      topicSlug: gathering.topicSlug,
      startDate: gathering.startDate,
      coverImageUrl: gathering.coverImageUrl,
    },
    score,
    reasons: explainRecommendation(user, gathering, score),
  }));
  
  return NextResponse.json({
    success: true,
    recommendations,
  });
}
```

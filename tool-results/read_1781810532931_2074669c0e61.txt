    55→### الملف: `src/lib/ai/llm-client.ts`
    56→
    57→```typescript
    58→import ZAI from 'z-ai-web-dev-sdk';
    59→
    60→// PROD-ONLY: استبدل بـ OpenAI client + API key management
    61→// الـ interface يبقى كما هو
    62→
    63→let zaiClient: Awaited<ReturnType<typeof ZAI.create>> | null = null;
    64→
    65→async function getClient() {
    66→  if (!zaiClient) {
    67→    zaiClient = await ZAI.create();
    68→  }
    69→  return zaiClient;
    70→}
    71→
    72→// In-memory cache (session-scoped)
    73→// PROD-ONLY: استبدل بـ Redis أو Upstash
    74→const cache = new Map<string, { value: string; expiresAt: number }>();
    75→
    76→const CACHE_TTL = 1000 * 60 * 30; // 30 minutes
    77→
    78→function getCached(key: string): string | null {
    79→  const entry = cache.get(key);
    80→  if (entry && entry.expiresAt > Date.now()) {
    81→    return entry.value;
    82→  }
    83→  cache.delete(key);
    84→  return null;
    85→}
    86→
    87→function setCached(key: string, value: string): void {
    88→  cache.set(key, { value, expiresAt: Date.now() + CACHE_TTL });
    89→}
    90→
    91→export interface LLMOptions {
    92→  systemPrompt?: string;
    93→  temperature?: number;
    94→  maxTokens?: number;
    95→  cacheKey?: string; // إن وُجد، استخدم الكاش
    96→  fallback?: string; // نص بديل لو فشل الـ LLM
    97→}
    98→
    99→export async function generateText(prompt: string, options: LLMOptions = {}): Promise<string> {
   100→  const {
   101→    systemPrompt = 'أنت مساعد ذكي لمنصة "لَمَّة" الكويتية للتجمعات الحميمية. تجيب بإيجاز وبأسلوب راقٍ يحترم الثقافة الكويتية.',
   102→    temperature = 0.7,
   103→    maxTokens = 500,
   104→    cacheKey,
   105→    fallback = 'عذراً، تعذّر توليد الرد حالياً. حاول مرة أخرى لاحقاً.',
   106→  } = options;
   107→  
   108→  // Check cache
   109→  if (cacheKey) {
   110→    const cached = getCached(cacheKey);
   111→    if (cached) return cached;
   112→  }
   113→  
   114→  try {
   115→    const client = await getClient();
   116→    const response = await client.chat.completions.create({
   117→      messages: [
   118→        { role: 'system', content: systemPrompt },
   119→        { role: 'user', content: prompt },
   120→      ],
   121→      temperature,
   122→      max_tokens: maxTokens,
   123→    });
   124→    
   125→    const result = response.choices[0]?.message?.content ?? fallback;
   126→    
   127→    // Cache it
   128→    if (cacheKey && result !== fallback) {
   129→      setCached(cacheKey, result);
   130→    }
   131→    
   132→    return result;
   133→  } catch (error) {
   134→    console.error('[LLM] Error:', error);
   135→    return fallback;
   136→  }
   137→}
   138→
   139→export async function generateJSON<T>(
   140→  prompt: string,
   141→  options: LLMOptions = {}
   142→): Promise<T | null> {
   143→  const result = await generateText(prompt, {
   144→    ...options,
   145→    systemPrompt: options.systemPrompt ?? 'أنت مساعد ذكي. أرجع النتيجة بصيغة JSON صحيحة فقط، بدون نص إضافي أو markdown.',
   146→  });
   147→  
   148→  try {
   149→    // Try to extract JSON from response
   150→    const jsonMatch = result.match(/\{[\s\S]*\}|\[[\s\S]*\]/);
   151→    if (jsonMatch) {
   152→      return JSON.parse(jsonMatch[0]) as T;
   153→    }
   154→    return JSON.parse(result) as T;
   155→  } catch (e) {
   156→    console.error('[LLM] JSON parse error:', e);
   157→    return null;
   158→  }
   159→}
   160→
   161→// Helper: build locale-aware prompt
   162→export function withLocale(locale: 'ar' | 'en', prompt: string): string {
   163→  return locale === 'ar'
   164→    ? `${prompt}\n\nأجب باللغة العربية الفصحى بأسلوب راقٍ.`
   165→    : `${prompt}\n\nRespond in English with an elegant editorial tone.`;
   166→}

   547→- يستخدم Zod للتحقق من المدخلات
   548→- يرجع `{ success: boolean, ... }`

   550→
   551→مثال (concierge):
   552→
   553→```typescript
   554→// src/app/api/v1/ai/concierge/route.ts
   555→import { NextRequest, NextResponse } from 'next/server';
   556→import { z } from 'zod';
   557→import { askConcierge } from '@/lib/ai/services/concierge';
   558→import { demoUsers } from '@/data/demo-users';
   559→
   560→const BodySchema = z.object({
   561→  question: z.string().min(3).max(500),
   562→  userId: z.string().optional(),
   563→  locale: z.enum(['ar', 'en']).default('ar'),
   564→});
   565→
   566→export async function POST(req: NextRequest) {
   567→  try {
   568→    const body = await req.json();
   569→    const { question, userId, locale } = BodySchema.parse(body);
   570→    
   571→    const user = userId ? demoUsers.find(u => u.id === userId) ?? null : null;
   572→    
   573→    const answer = await askConcierge(question, { user, locale });
   574→    
   575→    return NextResponse.json({
   576→      success: true,
   577→      answer,
   578→      timestamp: new Date().toISOString(),
   579→    });
   580→  } catch (error) {
   581→    return NextResponse.json(
   582→      { success: false, error: 'Invalid request' },
   583→      { status: 400 }
   584→    );
   585→  }
   586→}
   587→```
